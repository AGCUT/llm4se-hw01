### TripPlan（AI 生成结果?

```typescript
{
  title: string           // 行程标题
  destination: string     // 目的?
  startDate: string       // 开始日?
  endDate: string         // 结束日期
  days: number           // 天数
  budget: number         // 预算
  actualBudget: number   // 实际预算
  travelers: number      // 人数
  overview: {            // 概览
    highlights: string[]  // 亮点
    tips: string[]       // 建议
    summary: string      // 概述

  dailyPlans: DayPlan[] // 每日计划
  budgetBreakdown: {    // 预算分解
    transportation: number
    accommodation: number
    food: number
    attractions: number
    other: number
  }
}
```

```typescript
{
```
